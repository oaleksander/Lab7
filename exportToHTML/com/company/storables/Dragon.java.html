23       /**
24        * Converts CSV format string to Dragon
25        *
26        * @param csvString CSV format string
27        * @throws IllegalArgumentException if CSV format string can't be parsed
28        */
29       public Dragon(String csvString) throws IllegalArgumentException {
30           String[] splitString = csvString.split(",");
31           try {
32               if (splitString[0].isBlank())
33                   setId(DragonUtils.getNewId());
34               else
35                   try {
36                       if (Long.parseLong(splitString[0]) == -1L)
37                           setId(DragonUtils.getNewId());
38                       setId(Long.parseLong(splitString[0]));
39                   } catch (NumberFormatException e) {
40                       throw new IllegalArgumentException("Can't parse Dragon id from " + splitString[0] + ".");
41                   }
42               setName(splitString[1].replaceAll("\"", ""));
43               try {
44                   setCoordinates(new Coordinates(splitString[2].isBlank() ? .0 : Double.parseDouble(splitString[2]), Long.parseLong(splitString[3])));
45               } catch (NumberFormatException | NullPointerException e) {
46                   throw new IllegalArgumentException("Can't parse Dragon coordinates from " + splitString[2] + "," + splitString[3] + ".");
47               }
48               if (splitString[4].isBlank())
49                   setCreationDate(new Date());
50               else
51                   try {
52                       setCreationDate(DateFormat.getDateInstance(DateFormat.SHORT, Locale.getDefault()).parse(splitString[4]));
53                   } catch (ParseException e) {
54                       throw new IllegalArgumentException("Can't parse creation date from " + splitString[4] + ".");
55                   }
56               try {
57                   setAge(Long.parseLong(splitString[5]));
58               } catch (NumberFormatException e) {
59                   throw new IllegalArgumentException("Can't parse Dragon age from " + splitString[5] + ".");
60               }
61               setDescription(splitString[6].isBlank() ? "\"\"" : splitString[6].replaceAll("\"", ""));
62               try {
63                   setWeight(Integer.parseInt(splitString[7]));
64               } catch (NumberFormatException e) {
65                   throw new IllegalArgumentException("Can't parse Dragon weight from " + splitString[7] + ".");
66               }
67               try {
68                   setType(DragonType.valueOf(splitString[8]));
69               } catch (IllegalArgumentException e) {
70                   throw new IllegalArgumentException("Can't parse Dragon type from " + splitString[8] + ".");
71               }
72               setHead(new DragonHead(splitString[9]));
73               setOwner(splitString[10].isBlank() ? "\"\"" : splitString[10].replaceAll("\"", ""));
74           } catch (IndexOutOfBoundsException e) {
75               throw new IllegalArgumentException("Not enough arguments to parse Dragon (" + splitString.length + ").");
76           }
77   
78       }
