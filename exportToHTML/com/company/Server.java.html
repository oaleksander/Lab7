42           BufferedReader localInput = new BufferedReader(new InputStreamReader(System.in));
43           Selector selector = null;
44           ByteBuffer byteBuffer = ByteBuffer.allocate(65536);
45           try {
46               selector = Selector.open();
47               selector.wakeup();
48               DatagramChannel datagramChannel = DatagramChannel.open();
49               datagramChannel.bind(new InetSocketAddress("localhost", 3333));
50               datagramChannel.configureBlocking(false);

52           } catch (IOException e) {
53               System.err.println("Failed to start server.");
54               e.printStackTrace();

60               while (true) {
61                   try {
62                       String line;
63                       if (localInput.ready())
64                           if ((line = localInput.readLine()) != null)
65                               localExecutor.executeCommand(CommandReader.readCommandFromString(internalUser,line));
66                       selector.select(1500);
67                       selector.selectedKeys().stream().parallel().forEach(selectionKey ->
68                       {
69                           try {
70                               if (selectionKey.isValid())
71                                   if (selectionKey.isReadable())
72                                       responseExecutorService.submit(processCommand(selectionKey));
73                           } catch (IOException e) {
74                               System.err.println("IOException " + e.getMessage());
75                           } catch (NullPointerException ignored) {}
76                       });
77                       Executors.newCachedThreadPool().submit(() -> {

81                   } catch (Exception e) {

92       /**
93        * Processes a command from client
94      //  * @param buffer Byte buffer to put data to
95        * @param key Selection key
96        * @throws IOException If problem occurred while sending data to client
97        * @return Server response
98        */
99       private static Runnable processCommand(SelectionKey key)
100              throws IOException {
101          DatagramChannel client = (DatagramChannel) key.channel();
102          ByteBuffer buffer = ByteBuffer.allocate(65536);
103          SocketAddress address = client.receive(buffer);
104          if (address == null) throw new NullPointerException("Address is null");
105          byte[] data = buffer.array();
106          buffer.clear();
107          ObjectInputStream iStream = new ObjectInputStream(new ByteArrayInputStream(data));
108          byte[] response;
109          try {
110              CommandReader.Command command = (CommandReader.Command) iStream.readObject();
111              System.out.println("[" + new SimpleDateFormat("HH:mm:ss").format(new Date()) + "] " + command.user.getUsername() + address.toString() + ": " + command.toString() + ".");
112              //Отправляем данные клиенту
113              try {
114                  userExecutor.executeCommand(command);
115                  response = outputStream.toByteArray();
116                  outputStream.reset();
117              } catch (IllegalArgumentException e) {
118                  response = e.getMessage().getBytes();
119              }
120          } catch (StreamCorruptedException | ClassNotFoundException e) {
121              System.err.println(e.getMessage());
122              e.printStackTrace();
123              response = "Server received invalid packet. Please try again.".getBytes();
124          }
125          byte[] finalResponse = response;
126          return ()-> {
127              try {
128                  client.send(ByteBuffer.wrap(finalResponse), address);
129              } catch (IOException e) {
130                  System.err.println("IOException " + e.getMessage());
131              }
132          };
133      }
