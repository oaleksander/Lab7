128      /**
129       * Execute specified user command
130       *
131       * @param command User command
132       */
133      public void executeCommand(CommandReader.Command command) {
134          AtomicReference<String> response = new AtomicReference<>("Command gave no response.");
135          if(registeredUsers.stream().noneMatch(user -> user.equals(command.user)) && !command.commandString.equals("register")) {
136              response.set("Unauthorized access denied.");
137          } else {
138              if (Arrays.stream(availableCommands).parallel().noneMatch(availableCommand -> availableCommand.getLabel().equals(command.commandString))) {
139              response.set("Unknown command \"" + command.commandString + "\". try \"help\" for list of commands");
140          } else
141              try {
142                  Arrays.stream(availableCommands).forEach(availableCommand -> {
143                      if (command.commandString.equals(availableCommand.getLabel()))
144                          response.set(availableCommand.execute(command.user, command.argumentString));
145                  });
146              } catch (IllegalArgumentException e) {
147                  response.set(e.getMessage());
148              } catch (Exception e) {
149                  response.set("Unexpected error: " + e.getMessage() + ". This is a bug!");
150                  e.printStackTrace();
151              }
152          }
153          printStream.println(response.get());
154      }
